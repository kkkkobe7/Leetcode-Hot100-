# 148. 排序链表（归并+快排）

归并排序代码：

```c++
class Solution {
public:
    ListNode* merge(ListNode* a, ListNode* b)
    {
        ListNode *ans = new ListNode();
        ListNode *p = a, *q = b;
        while (p && q)
        {
            if (p->val <= q->val) 
            {
                ans->next = p;
                p = p->next;
            }
            else
            {
                ans->next = q;
                q = q->next;
            }
            ans = ans->next;
        }
        if (p) ans->next = p;
        if (q) ans->next = q;
        return a->val <= b->val? a: b;
    }

    ListNode* sortList(ListNode* head) {
        if (head == nullptr || head->next == nullptr) return head;

        ListNode *slow = head, *fast = head, *la = head; // 找到中间的节点，一分而二
        while (fast && fast->next)
        {
            la = slow;
            slow = slow->next;
            fast = fast->next->next;
        }
        
        la->next = nullptr; // 一分为二
        ListNode* l = sortList(head);
        ListNode* r = sortList(slow);

        return merge(l, r);
    }
};
```



快排代码（时间复杂度不如归并，会卡数据点，但面试可能会考）：

```c++
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        // 1. 递归终止：空或只有一个节点
        if (!head || !head->next) return head;

        // 2. 三个哨兵节点，lp, mp, rp 分别是三条链表的当前尾指针
        ListNode L(0), M(0), R(0);
        ListNode *lp = &L, *mp = &M, *rp = &R;
        
        int pivot = head->val;
        // 3. 一次遍历，将节点分发到三个篮子里
        for (ListNode* p = head; p; p = p->next) {
            if (p->val < pivot)      lp = lp->next = p;
            else if (p->val > pivot) rp = rp->next = p;
            else                     mp = mp->next = p;
        }

        // 4. 关键：切断三条链表的末尾，防止成环
        lp->next = mp->next = rp->next = nullptr;

        // 5. 递归排序左右两部分
        L.next = sortList(L.next);
        R.next = sortList(R.next);

        // 6. 串联：L -> M -> R
        // 先找 L 的尾巴（sorted L 之后的末尾）
        ListNode* tail = &L;
        while (tail->next) tail = tail->next;
        
        // 拼接：L尾 -> M头；M尾 -> R头
        tail->next = M.next;
        mp->next = R.next;

        return L.next; // 返回 L 的头
    }
};
```
