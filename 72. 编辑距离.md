# 72. 编辑距离

解题思路拆解

**1. 核心模型：寻找子问题的关联**
*   假设我们要算 `horse` 变成 `ros` 的距离。如果我们已经知道 `hors` 变成 `ro` 的距离，能不能推导出下一步？
*   **状态定义**：`dp[i][j]` 表示 `word1` 的前 `i` 个字符 变成 `word2` 的前 `j` 个字符 所需的最少步数。

**2. 状态转移逻辑（核心推导）**
观察 `word1` 的第 $i$ 个字符（记作 `A`）和 `word2` 的第 $j$ 个字符（记作 `B`）：

*   **情况 A：字符相等 (`word1[i-1] == word2[j-1]`)**
    *   既然末尾已经一样了，我们不需要做任何操作。
    *   `dp[i][j] = dp[i-1][j-1]`。
*   **情况 B：字符不相等**
    我们需要从三种操作中选一个代价最小的：
    1.  **替换**：把 `A` 换成 `B`。那么步数等于之前的 `dp[i-1][j-1]` 再加 1 步。
    2.  **删除**：把 `word1` 的 `A` 删掉。那么我们需要看 `word1` 前 $i-1$ 个字符变成 `word2` 前 $j$ 个字符需要多少步。公式：`dp[i-1][j] + 1`。
    3.  **插入**：在 `word1` 末尾插入一个 `B`。那么 `word1` 原有的 $i$ 个字符只需要匹配 `word2` 前 $j-1$ 个字符即可。公式：`dp[i][j-1] + 1`。

**3. 初始状态 (Base Case)**
*   如果 `word1` 是空的（长度 0），要变成长度为 $j$ 的 `word2`，只能不停地**插入**，步数就是 $j$。
*   如果 `word2` 是空的，要变成长度为 0，只能不停地**删除**，步数就是 $i$。



简洁版代码：

```c++
class Solution {
public:
    static const int N = 510;
    // dp[i][j] 代表 word1 的前 i 个字符转换成 word2 的前 j 个字符的最少操作数
    int dp[N][N]; 

    int minDistance(string word1, string word2) {
        int n = word1.size(), m = word2.size();

        if (n == 0 || m == 0) return max(n, m);

        for (int j = 0; j <= m; j++) dp[0][j] = j;
        for (int i = 0; i <= n; i++) dp[i][0] = i;

        for (int i = 1; i <= n; i ++)
        {
            for (int j = 1; j <= m; j ++)
            {
                if (word1[i-1] == word2[j-1]) dp[i][j] = dp[i-1][j-1];
                else
                {
                    // 字符不等，取三种操作的最小值：
                    // dp[i-1][j-1] + 1 : 替换
                    // dp[i][j-1] + 1   : 插入
                    // dp[i-1][j] + 1   : 删除
                    dp[i][j] = min({dp[i-1][j-1], dp[i][j-1], dp[i-1][j]}) + 1;
                }
            }
        }
        
        return dp[n][m];
    }
};
```

注释版代码：

```c++
class Solution {
public:
    static const int N = 510;
    // dp[i][j] 代表 word1 的前 i 个字符转换成 word2 的前 j 个字符的最少操作数
    int dp[N][N]; 

    int minDistance(string word1, string word2) {
        int n = word1.size(), m = word2.size();

        // 1. 特判与初始化基础情况
        if (n == 0 || m == 0) return max(n, m);

        // 初始化第一行：word1 是空串时，变成 word2 需要的插入操作数
        for (int j = 0; j <= m; j++) dp[0][j] = j;
        // 初始化第一列：word2 是空串时，word1 需要的删除操作数
        for (int i = 0; i <= n; i++) dp[i][0] = i;

        // 2. 动态规划填表
        for (int i = 1; i <= n; i ++)
        {
            for (int j = 1; j <= m; j ++)
            {
                // 注意：由于 dp 下标从 1 开始，对应字符串下标要减 1
                if (word1[i-1] == word2[j-1]) 
                {
                    // 字符相等，不增加操作数，直接继承前一个状态
                    dp[i][j] = dp[i-1][j-1];
                }
                else
                {
                    // 字符不等，取三种操作的最小值：
                    // dp[i-1][j-1] + 1 : 替换
                    // dp[i][j-1] + 1   : 插入
                    // dp[i-1][j] + 1   : 删除
                    dp[i][j] = min({dp[i-1][j-1], dp[i][j-1], dp[i-1][j]}) + 1;
                }
            }
        }
        
        // 最终结果存储在矩阵右下角
        return dp[n][m];
    }
};
```
