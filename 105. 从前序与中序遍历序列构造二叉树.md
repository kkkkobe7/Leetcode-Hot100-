# 105. 从前序与中序遍历序列构造二叉树

解题思路拆解

**1. 理解遍历序列的“基因”特征**
*   **前序遍历 (Preorder)**：顺序是 `[ 根节点 | 左子树 | 右子树 ]`。
    *   **核心规律**：数组的**第一个元素**永远是当前树的根节点。
*   **中序遍历 (Inorder)**：顺序是 `[ 左子树 | 根节点 | 右子树 ]`。
    *   **核心规律**：只要我们在中序遍历中找到了根节点的位置，它的**左边**全是左子树，**右边**全是右子树。

**2. 寻找拆解公式（分治思想）**
1.  **找根**：从前序遍历中拿到第一个数作为当前根。
2.  **定界**：在中序遍历中找到这个根的位置（假设下标为 `idx`）。
3.  **算大小**：计算左子树的节点数量 `len = idx - 开始位置`。
4.  **递归**：
    *   **左子树的前序范围**：根节点后面连续的 `len` 个数。
    *   **左子树的中序范围**：根节点左边的部分。
    *   **右子树的前序范围**：剩下的那部分。
    *   **右子树的中序范围**：根节点右边的部分。

**3. 递归终止条件**
*   当区间范围不合法（左边界 > 右边界）时，说明这一侧已经没有节点了，返回 `nullptr`。



简洁版代码：

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* build(vector<int>& pre, int l1, int r1, vector<int>& in, int l2, int r2)
    {
        if (l1 > r1 || l2 > r2) return nullptr;

        int root = pre[l1];
        TreeNode* cur = new TreeNode(root);
        int idx = l2;
        for (int i = l2; i <= r2; i ++)
        {
            if (in[i] == root)
            {
                idx = i;
                break;
            }
        }
        // 长度为 idx - l2
        cur->left = build(pre, l1 + 1, l1 + idx - l2, in, l2, idx - 1);
        cur->right = build(pre, l1 + idx - l2 + 1, r1, in, idx + 1, r2);

        return cur;
    }

    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        // preorder 根左右，inorder 左根右，每次根据preorder找到根，然后分别划分
        int n = preorder.size();
        TreeNode* root = build(preorder, 0, n - 1, inorder, 0, n - 1);
        return root;
    }
};
```

注释版代码：

```c++
class Solution {
public:
    // 辅函数：通过下标区间来构造树，避免频繁拷贝 vector
    // pre: 前序序列, [l1, r1] 范围
    // in : 中序序列, [l2, r2] 范围
    TreeNode* build(vector<int>& pre, int l1, int r1, vector<int>& in, int l2, int r2)
    {
        // 1. 递归终止条件：区间长度小于等于 0
        if (l1 > r1 || l2 > r2) return nullptr;

        // 2. 确定根节点
        // 前序遍历的第一个元素一定是当前的根
        int rootVal = pre[l1];
        TreeNode* cur = new TreeNode(rootVal);

        // 3. 在中序遍历中寻找根节点的位置
        // 目的：为了划分出左右子树的边界
        int idx = l2;
        for (int i = l2; i <= r2; i ++)
        {
            if (in[i] == rootVal)
            {
                idx = i;
                break;
            }
        }

        // 4. 计算左子树的长度
        // 左子树长度 = 中序遍历中根节点下标 - 中序遍历开始下标
        int leftLen = idx - l2;

        // 5. 递归构造左右子树
        // 左子树：
        // 前序跳过根节点开始，取 leftLen 长度：[l1 + 1, l1 + leftLen]
        // 中序取根节点左边：[l2, idx - 1]
        cur->left = build(pre, l1 + 1, l1 + leftLen, in, l2, idx - 1);

        // 右子树：
        // 前序从左子树结束后的下一个开始：[l1 + leftLen + 1, r1]
        // 中序取根节点右边：[idx + 1, r2]
        cur->right = build(pre, l1 + leftLen + 1, r1, in, idx + 1, r2);

        return cur;
    }

    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        int n = preorder.size();
        if (n == 0) return nullptr;
        
        // 传入初始边界 [0, n-1]
        return build(preorder, 0, n - 1, inorder, 0, n - 1);
    }
};
```
