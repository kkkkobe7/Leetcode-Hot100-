# 85. 最大矩形

解题思路拆解

**1. 核心转化：化面为线**
*   我们将矩阵的每一行看作是“地面”。
*   对于每一行，我们计算每一个位置向上连续的 '1' 的数量。这就像是在这一行地面上建立起了一个**直方图（Histogram）**。
*   **例子**：
    矩阵：
    ```
    1 0 1 0 0  -> 第一行直方图高度：[1, 0, 1, 0, 0]
    1 0 1 1 1  -> 第二行直方图高度：[2, 0, 2, 1, 1] (第一行累加)
    1 1 1 1 1  -> 第三行直方图高度：[3, 1, 3, 2, 2] (第二行累加)
    ```
*   **结论**：整个矩阵的最大矩形，必然是这些行所构成的某个直方图中的最大矩形。

**2. 算法流程**
1.  **预处理高度**：遍历矩阵，维护一个 `heights` 数组（或矩阵）。如果 `matrix[i][j] == '1'`，高度就是 `上一行高度 + 1`；如果为 '0'，高度直接断掉，设为 `0`。
2.  **逐行求解**：每处理完一行的高度，就调用一次 [LeetCode 84] 的解法（单调栈），算出当前直方图里的最大矩形面积。
3.  **更新全局最值**：记录所有行中出现过的最大面积。

**3. 单调栈回顾 (1D 求解器)**
*   对于每一行的 `heights` 数组，利用单调递增栈找到每个柱子左边和右边第一个比它矮的位置，从而确定该柱子能向两侧扩张的最大宽度。



简洁版代码：

```c++
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int n = heights.size(), ans = 0;

        stack<int> st;
        vector<int> l(n), r(n, n);

        for (int i = 0; i < n; i ++)
        {
            while (st.size() && heights[i] <= heights[st.top()]) 
            {
                r[st.top()] = i;
                st.pop();
            }
            
            l[i] = st.empty()? -1: st.top();
            st.push(i);
        }

        for (int i = 0; i < n; i ++) ans = max(ans, heights[i] * (r[i] - l[i] - 1));

        return ans;
    }

    int maximalRectangle(vector<vector<char>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        vector<vector<int>> heights(n, vector<int>(m, 0));
        int ans = 0;
        
        for (int i = 0; i < n; i ++)
        {
            for (int j = 0; j < m; j ++)
            {
                if (matrix[i][j] == '1') 
                {
                    heights[i][j] = 1;
                    if (i) heights[i][j] += heights[i - 1][j];
                }
            }
        }

        for (int i = 0; i < n; i ++) ans = max(ans, largestRectangleArea(heights[i]));

        return ans;
    }
};
```

注释版代码：

```c++
class Solution {
public:
    // ---------------------------------------------------------
    // 辅函数：求解 1D 直方图的最大矩形面积 (LeetCode 84 逻辑)
    // ---------------------------------------------------------
    int largestRectangleArea(vector<int>& heights) {
        int n = heights.size(), ans = 0;

        stack<int> st; // 单调递增栈，存储下标
        vector<int> l(n), r(n, n); // l 和 r 分别记录左右第一个比当前矮的边界

        for (int i = 0; i < n; i ++)
        {
            // 如果当前柱子比栈顶矮，栈顶的右边界就找到了
            while (st.size() && heights[i] <= heights[st.top()]) 
            {
                r[st.top()] = i;
                st.pop();
            }
            
            // 当前柱子的左边界就是它入栈时的栈顶
            l[i] = st.empty() ? -1 : st.top();
            st.push(i);
        }

        // 计算每一根柱子作为高度时的最大面积
        for (int i = 0; i < n; i ++) 
            ans = max(ans, heights[i] * (r[i] - l[i] - 1));

        return ans;
    }

    // ---------------------------------------------------------
    // 主函数：将 2D 矩阵问题转化为 1D
    // ---------------------------------------------------------
    int maximalRectangle(vector<vector<char>>& matrix) {
        if (matrix.empty()) return 0;
        int n = matrix.size(), m = matrix[0].size();
        
        // heights[j] 代表当前行中，第 j 列向上有多少个连续的 '1'
        vector<int> rowHeights(m, 0);
        int ans = 0;
        
        // 逐行扫描矩阵
        for (int i = 0; i < n; i ++)
        {
            for (int j = 0; j < m; j ++)
            {
                // 如果当前是 '1'，累加高度
                if (matrix[i][j] == '1') 
                {
                    rowHeights[j] += 1;
                }
                // 如果当前是 '0'，柱子断了，高度清零
                else 
                {
                    rowHeights[j] = 0;
                }
            }
            
            // 对当前这一行形成的“直方图”求最大矩形
            ans = max(ans, largestRectangleArea(rowHeights));
        }

        return ans;
    }
};
```
